    """

[docs]def top (s):
    """
    :param s: stack to pop
    :type s: stack
    :return: element on top of s without removing it
    :UC: s must be non empty
    """
    try:
        return s[-1]
    except IndexError:
        raise StackEmptyError('empty stack, nothing on the top')
